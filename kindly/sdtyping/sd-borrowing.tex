  }{

    \lregion{C}{x}{\E;\svar{x}{\tau}^n}{\E; b}
  }

    \BORROW\in\left\{\kun,\kaff\right\}\\

  }{
    \bregion{C}{}
    {\svar[\BORROW]{x}{\schm}^n}
    {\bbvar[\BORROW]{x} k{\schm}}
  }
